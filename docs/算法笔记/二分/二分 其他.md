```python

class Solution:

    def mySqrt(self, x: int) -> int:

        left,right = 0,x + 1

        while left + 1 < right:

            mid = left + (right - left) // 2

            if mid * mid <= x:

                left = mid

            else:

                right = mid

        return left

```
##  核心结论 (One-Liner)

> [!summary] 本题考点
> 越小越满足，更新left

class Solution:

    def findPeakElement(self, nums: List[int]) -> int:

        left,right = -1,len(nums) - 1

        while left + 1 < right:

            mid = left + (right - left) // 2

            if nums[mid] > nums[mid + 1]:

                right = mid

            else:

                left = mid

        return right

```
##  核心结论 (One-Liner)

> [!summary] 本题考点
> 有mid + 1下标，需要注意不要越界 避免越界判断的方法是right = len（nums） - 1

class Solution:

    def findPeakGrid(self, mat: List[List[int]]) -> List[int]:

        left,right = -1,len(mat) - 1

        while left + 1 < right:

            mid = left + (right - left) // 2

            mx = max(mat[mid])

            if mx > mat[mid + 1][mat[mid].index(mx)]:

                right = mid

            else:

                left = mid

        return [right,mat[right].index(max(mat[right]))]


```
