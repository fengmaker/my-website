```python
class Solution:

    def hIndex(self, citations: List[int]) -> int:

        left = 0

        right = len(citations) + 1

        while left + 1 < right:

            mid = left + (right - left) // 2

            if citations[-mid] >= mid:

                left = mid

            else:

                right = mid

        return left

```
##  核心结论 (One-Liner)

> [!summary] 本题考点
> 求最大，left check True right False return left

```python
class Solution:

    def maximumCandies(self, candies: List[int], k: int) -> int:

        left = 0

        right = min(max(candies),sum(candies) // k) + 1

        while left + 1 < right:

            mid = left + (right - left) // 2

            if sum(c // mid for c in candies) >= k:

                left = mid

            else:

                right = mid

        return left
```
##  核心结论 (One-Liner)

> [!summary] 本题考点
> 注意÷0错误，如果mid可以等于0要小心，当然最好0满足条件的话，就不要放在二分区间，有的时候可以优化区间下界和上届，可以注意

```python
class Solution:

    def maximumLength(self, s: str) -> int:

        gruops = defaultdict(list)

        cnt = 0

        for i,ch in enumerate(s):

            cnt += 1

            if i + 1 == len(s) or ch != s[i + 1]:

                gruops[ch].append(cnt)

                cnt = 0  

        ans = 0

        for a in gruops.values():

            a.sort(reverse = True)

            a.extend([0,0])

            ans = max(ans,a[0] - 2,min(a[0] - 1,a[1]),a[2])

        return ans if ans else -1


```

> [!summary] 本题考点
