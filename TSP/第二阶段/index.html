def subtour_callback(model, where):
    # GRB.Callback.MIPSOL Ë°®Á§∫Ê±ÇËß£Âô®ÊâæÂà∞‰∫Ü‰∏Ä‰∏™Êñ∞ÁöÑÊï¥Êï∞Ëß£
    if where == GRB.Callback.MIPSOL:

        # 1. Ëé∑ÂèñÂΩìÂâçÊï¥Êï∞Ëß£
        x_sol = model.cbGetSolution(model._vars)
        n = model._n

        # 2. ÊûÑÂª∫Âõæ (‰ªÖÂåÖÂê´Ë¢´ÈÄâ‰∏≠ÁöÑËæπ)
        G = nx.Graph()
        G.add_nodes_from(range(n))
        for (i, j) in x_sol.keys():
            if x_sol[i, j] > 0.5:
                G.add_edge(i, j)

        # 3. ÂØªÊâæËøûÈÄöÂàÜÈáè (Separation)
        components = list(nx.connected_components(G))

        # 4. Â¶ÇÊûúËøûÈÄöÂàÜÈáè > 1ÔºåËØ¥ÊòéÂ≠òÂú®Â≠êÂõûË∑Ø -> Ê∑ªÂä†Á∫¶Êùü
        if len(components) > 1:
            for component in components:
                # Á≠ñÁï•ÔºöÈíàÂØπÊØè‰∏™Â≠êÂõûË∑Ø SÔºåÊ∑ªÂä† DFJ Á∫¶Êùü
                # sum(x_ij for i,j in S) <= |S| - 1

                # ËøôÈáåÁöÑ quicksum ÂøÖÈ°ªÈÅçÂéÜ component ÂÜÖÈÉ®ÊâÄÊúâÂèØËÉΩÁöÑËæπ
                expr = gp.quicksum(model._vars[min(i,j), max(i,j)] 
                                   for i in component 
                                   for j in component 
                                   if i != j and (min(i,j), max(i,j)) in model._vars)

                # Ê∑ªÂä† Lazy Constraint (ËøôÂ∞±Áõ∏ÂΩì‰∫éÂä®ÊÄÅÂàá‰∫Ü‰∏ÄÂàÄ)
                model.cbLazy(expr <= len(component) - 1)

    # ËÆ°ÁÆóË∑ùÁ¶ªÁü©Èòµ (Áï•ÔºåÂÅáËÆæÂ∑≤Êúâ)
    dist = {(i,j): math.dist(points[i], points[j]) for i in range(n) for j in range(i+1, n)}

    m = gp.Model("TSP_BCP")

    # ÂèòÈáèÔºöx[i,j] ‰∏∫‰∫åËøõÂà∂ÂèòÈáè
